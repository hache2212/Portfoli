DELIMITER //

CREATE PROCEDURE list_clubs_by_country(IN country_name VARCHAR(20))

BEGIN

        DECLARE done INT DEFAULT 0;

        DECLARE name_club VARCHAR(40);

        DECLARE cur CURSOR FOR SELECT name_club FROM club WHERE country = country_name;

        DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;

        OPEN cur;

        read_loop: LOOP

                FETCH cur INTO name_club;

                IF done THEN

                        LEAVE read_loop;

                END IF;

                SELECT name_club;

        END LOOP;

        CLOSE cur;

END //

DELIMITER ;

CALL list_clubs_by_country('Spain');

CREATE PROCEDURE update_player_awards(IN player_id INT(4))

BEGIN

  DECLARE awards_received INT;

  SELECT COUNT(*) INTO awards_received FROM player_awards WHERE id_award = player_id;

  UPDATE player SET awards = awards_received WHERE id_player = player_id;

END //

DELIMITER ;

CALL update_player_awards(10);

SELECT id_player, name_player, awards FROM player WHERE id_player = 10;

CREATE PROCEDURE add_player(IN name VARCHAR(40), IN country VARCHAR(20), IN age INT(2), IN position VARCHAR(20), IN club_id INT(4), OUT new_player_id INT(4))

BEGIN

        DECLARE EXIT HANDLER FOR SQLEXCEPTION

        BEGIN

                ROLLBACK;

                SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Error adding player';

        END;

        START TRANSACTION;

                INSERT INTO player (name_player, country, age, position, club_id, coach_id, sports_director_id) VALUES (name, country, age, position, club_id, NULL, NULL);

                SET new_player_id = LAST_INSERT_ID();

        COMMIT;

    SELECT id_player, name_player, country, age, position, club_id, coach_id, sports_director_id FROM player WHERE id_player = new_player_id;

END //

DELIMITER ;

CALL add_player('Martin Odegaard', 'Norway', 23, 'Midfielder', 8, @new_player_id);

SELECT @new_player_id AS new_player_id;

CREATE PROCEDURE update_player_salary(IN player_id INT(4), IN goals_scored INT)

BEGIN

    DECLARE new_salary DECIMAL(10,2);

   

    SET new_salary = (SELECT salary FROM contract WHERE player_id = player_id) * (1 + goals_scored * 0.05);

   

    UPDATE contract SET salary = new_salary WHERE player_id = player_id;

END //

DELIMITER ;

CALL update_player_salary(1, 10);

Task 5. Documentation

Create the appropriate documentation to explain the actions taken and what each object created, removed or modified from the web application is for.

  1. avg_age_of_players procedure: This stored procedure calculates the average age of players in a given club. It takes the club ID as an input parameter and returns the calculated average age as an output parameter.
  2. add_stadium procedure: This stored procedure adds a new stadium to the stadium table. It takes the name, city, country, capacity, and year of creation as input parameters and returns the generated stadium ID as an output parameter.
  3. update_club_stadium procedure: This stored procedure updates the stadium ID of a club in the club table. It takes the club name and stadium ID as input parameters.
  4. list_clubs_by_country procedure: This stored procedure lists all clubs from a specific country. It takes the country name as an input parameter and returns a result set containing the club names.
  5. update_player_awards procedure: This stored procedure updates the number of awards received by a player in the player table. It takes the player ID as an input parameter.
  6. update_player_information procedure: This stored procedure updates the age, position, and club ID of a player in the player table. It takes the player ID, new age, new position, and new club ID as input parameters.
  7. add_player procedure: This stored procedure adds a new player to the player table. It takes the name, country, age, position, and club ID as input parameters and returns the generated player ID as an output parameter.
  8. assign_agent_to_player procedure: This stored procedure assigns an agent to a player in the player table. It takes the player ID and agent ID as input parameters.
  9. assign_coach_to_sports_director procedure: This stored procedure assigns a coach to a sports director in the coach table. It takes the coach ID and sports director ID as input parameters.
  10. update_player_salary procedure: This stored procedure updates the salary of a player in the contract table based on their performance (e.g., number of goals scored). It takes the player ID and the number of goals scored as input parameters.
